def split_data(df: pd.DataFrame) -> tuple[pd.DataFrame, pd.Series]:
    """
    Split a DataFrame into features and target.

    Parameters
    ----------
    df : pd.DataFrame
        Full dataset containing a column named 'target'.

    Returns
    -------
    X : pd.DataFrame
        All columns except 'target'.
    y : pd.Series
        The 'target' column.
    """
    X = df.drop("target", axis=1)
    y = df["target"]
    return X, y
